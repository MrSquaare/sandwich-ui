---

## Using the styled system

```jsx
import { css } from "@sandwich-ui/core/css";

function Example() {
  return (
    <p className={css({ fontSize: "lg" })}>Lorem ipsum</p>
  );
}
```

## Using a component

```jsx
import { Button } from "@sandwich-ui/react";

function Example() {
  return (
    <Button>Click me</Button>
  );
}
```
